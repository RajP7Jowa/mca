    Four scalar types: (1)bool,    (2)int,    (3)float,    (4)string 

    
<?php
    $a_bool = TRUE;   // a boolean
    $a_str  = "foo";  // a string
    $a_str2 = 'foo';  // a string
    $an_int = 12;     // an integer
    $b_floating = 1.1354 //a float

    echo gettype($a_bool); // prints out:  boolean
    echo gettype($a_str);  // prints out:  string


    // If this is an integer, then print it out
    if (is_int($an_int)) {
        echo 'Yes $an_int is integer';
    }

    // If $a_bool is a string, print it out
    if (is_string($a_bool)) {
        echo "String: $a_bool";
    }

?>

<?php

    echo "$var, $Var";      // outputs "Bob, Joe"

    $4site = 'not yet';     // invalid; starts with a number
    $_4site = 'not yet';    // valid; starts with an underscore
    $täyte = 'mansikka';    // valid; start with String or Character.

    $foo = new foo();
    $bar = 'bar';

    echo $foo->$bar . "\n";

  • Variable Scope

    
<?php

$a_glbl = "globalVariable
    ";

function funFoo()
{
    global $a_glbl; //redefined to global Variable
    echo $a_glbl;   //use global Variable 
    $a_glbl ="ChangeGlobal
    ";

    $b_lcl ="localVariable
    "; //defined local Variable
    echo $b_lcl; // use local Variable
}


echo $a_glbl; //before call funFoo print globalVariable
funFoo();

//after call funFoo print
echo $a_glbl; //ChangeGlobal because funFoo change the value of variable
?>
